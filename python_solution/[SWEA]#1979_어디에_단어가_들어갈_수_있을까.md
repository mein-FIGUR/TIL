def check_word_len(puzzle, n, k):
    result = 0

    for line in puzzle:
        start, idx = 0, 0
        flag = 0
        
        while idx < n:
            # 0->1
            if line[idx] and not flag :
                start = idx
                flag = 1
            #1->0
            elif not line[idx] and flag :
                if idx - start == k :
                    result += 1
                flag = 0
            
            idx += 1
            
        if flag and idx - start == k:
            result += 1
