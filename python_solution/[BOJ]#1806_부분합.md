N, S = map(int, input().split())
numlist = list(map(int, input().split()))

result = N+1
start, end = 0, 1
now = numlist[start]

while True :
    if now >= S :
        result = min(result, end-start)
        now -= numlist[start]
        start += 1
    elif end == N :
        break
    else :
        now += numlist[end]
        end += 1
    
if result == N+1 :
    result = 0
    
print(result)
