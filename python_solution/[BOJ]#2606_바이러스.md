n = int(input())
e = int(input())
connected = {i: [] for i in range(1, n+1)}
for _ in range(e):
    n1, n2 = map(int, input().split())
    connected[n1].append(n2)
    connected[n2].append(n1)

visited = [0]*n


def dfs(v):
    visited[v-1] = 1
    for node in connected[v]:
        if not visited[node-1]:
            dfs(node)


dfs(1)
print(visited.count(1)-1)
